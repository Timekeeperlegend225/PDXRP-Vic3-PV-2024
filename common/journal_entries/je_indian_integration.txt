je_indian_removal_ac = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"
	group = je_group_internal_affairs
	is_shown_when_inactive = {
		has_technology_researched = colonization
		NOT = {
			c:DKT ?= THIS
		}
		NOT = {
			c:LOB ?= THIS
		}
		NOT = {
			c:YUA ?= THIS
		}
	}
	scripted_button = je_ind_rem_but
	possible = {
		OR = {
			any_scope_state = {
				count >= 1
				AND = {
					region = sr:region_new_england
					owner = ROOT
				}
			}
			any_scope_state = {
				count >= 1
				AND = {
					region = sr:region_the_midwest
					owner = ROOT
				}
			}
			any_scope_state = {
				count >= 1
				AND = {
					region = sr:region_canada
					owner = ROOT
				}
			}
			any_scope_state = {
				count >= 1
				AND = {
					region = sr:region_pacific_coast
					owner = ROOT
				}
			}
			any_scope_state = {
				count >= 1
				AND = {
					region = sr:region_great_plains
					owner = ROOT
				}
			}
			any_scope_state = {
				count >= 1
				AND = {
					region = sr:region_dixie
					owner = ROOT
				}
			}
		}
	}
	immediate = {
		trigger_event = {
			id = fm_ind.1
		}
		random_scope_culture = {
			limit = {
				is_primary_culture_of = root
			}
			save_scope_as = ind_int_culture
		}
	}
	timeout = 1825
	complete = {
		custom_tooltip = {
			text = tooltip_has_indrem
			has_variable = variable_indi_app
		}
	}
	on_complete = {
		add_modifier = {
			name = modifier_ind_two
		}
		trigger_event = {
			id = fm_ind.2
		}
	}
	on_timeout = {
		add_modifier = {
			name = modifier_ind_one
		}
		trigger_event = {
			id = fm_ind.3
		}
	}
	weight = 100
	should_be_pinned_by_default = yes
}
